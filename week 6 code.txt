#include <WiFi.h>

#include <LiquidCrystal_I2C.h>
#include <ESP32Servo.h>
#include <Stepper.h>

// WiFi
char ssid[] = "PISO WIFI PASSWORD";
char pass[] = "123123123";

// Ultrasonic
#define TRIG_PIN 5
#define ECHO_PIN 18

// LED
#define LED_PIN 4
#define DIST_THRESHOLD 20  // cm

// LCD
LiquidCrystal_I2C lcd(0x27,16,2);

// Servos
Servo servo1;
Servo servo2;

#define SERVO1_PIN 19
#define SERVO2_PIN 23

// Stepper Motor
#define STEPS 2048
Stepper stepper(STEPS, 13, 12, 14, 27);

BlynkTimer timer;

// Flag so cleaning runs only once
bool objectDetected = false;

// -------- ULTRASONIC + LED + LCD + AUTO MOTORS --------
void sendSensor()
{
  // Trigger ultrasonic
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  // Read echo
  long duration = pulseIn(ECHO_PIN, HIGH);
  float distance = duration * 0.034 / 2;

  int ledState = 0;

  // ---------- OBJECT DETECTED ----------
  if(distance <= DIST_THRESHOLD && distance > 0)
  {
    digitalWrite(LED_PIN, HIGH);
    ledState = 1;

    // Run cleaning motion only once
    if(!objectDetected)
    {
      objectDetected = true;

      // Cleaning motion (open-close)
      for(int i = 0; i < 3; i++)
      {
        servo1.write(90);   // open
        servo2.write(90);
        delay(300);

        servo1.write(0);    // close
        servo2.write(0);
        delay(300);
      }

      // Stepper motor rotate forward
      stepper.step(200);
    }
  }

  // ---------- NO OBJECT ----------
  else
  {
    digitalWrite(LED_PIN, LOW);
    ledState = 0;

    objectDetected = false;

    // Return servos to initial position
    servo1.write(0);
    servo2.write(0);

    // Stepper rotate backward
    stepper.step(-200);
  }

  // ---------- SEND TO BLYNK ----------
  Blynk.virtualWrite(V0, distance);
  Blynk.virtualWrite(V1, ledState);

  // ---------- LCD DISPLAY ----------
  lcd.setCursor(0,0);
  lcd.print("Distance:      ");
  lcd.setCursor(0,0);
  lcd.print(distance);
  lcd.print(" cm");

  lcd.setCursor(0,1);
  if(ledState)
    lcd.print("Object Detected");
  else
    lcd.print("No Object      ");

  // Serial Monitor
  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.print(" cm  LED: ");
  Serial.println(ledState);
}

// -------- SETUP --------
void setup()
{
  Serial.begin(115200);

  Wire.begin(21,22); // LCD I2C pins

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);

  // LCD
  lcd.init();
  lcd.backlight();

  // Servos
  servo1.attach(SERVO1_PIN);
  servo2.attach(SERVO2_PIN);

  // Stepper
  stepper.setSpeed(10);

  // Connect Blynk
  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass);

  // Read sensor every 2 seconds
  timer.setInterval(2000L, sendSensor);
}

// -------- LOOP --------
void loop()
{
  Blynk.run();
  timer.run();
}
